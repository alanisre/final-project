<script>
function generateWorkout() {
  let workouts = [
    "10 Push-ups",
    "20 Squats",
    "30 sec Plank",
    "15 Lunges",
    "20 Jumping Jacks"
  ];

  let random = workouts[Math.floor(Math.random() * workouts.length)];
  document.getElementById("result").innerText = random;
}
</script>
